from app.ingest.chunker import Chunk

    If the row exists with a matching content_hash, return the existing UUID untouched.
    If the row exists with a different content_hash, UPDATE content_hash and fetched_at, return existing UUID.
    Returns documents.id as a UUID string.

    Uses ON CONFLICT (source_type, source_id) DO UPDATE ... RETURNING id.
    """
    ...

def chunks_to_embed(
    conn: psycopg.Connection,
    document_id: str,
    chunks: list[Chunk],
    embedding_model: str,
) -> tuple[list[Chunk], list[Chunk]]:
    """
    Partition `chunks` into (to_embed, to_skip) by checking existing rows in `chunks` table.

    A chunk is in `to_skip` iff a row with the SAME `(document_id, chunk_index, embedding_model)`
    AND the SAME `content_hash` already exists.

    A chunk is in `to_embed` iff:
      - no row with that (document_id, chunk_index, embedding_model) exists, OR
      - a row exists but its content_hash differs (the chunk text changed — re-embed).
    """
    ...

def upsert_chunks(
    conn: psycopg.Connection,
    document_id: str,
    chunks: list[Chunk],
    vectors: list[list[float]],
    embedding_model: str,
) -> int:
    """
    Insert `chunks` (with vectors aligned by position) into the `chunks` table.

    Uses ON CONFLICT (document_id, chunk_index, embedding_model) DO UPDATE
      SET chunk_text = EXCLUDED.chunk_text,
          content_hash = EXCLUDED.content_hash,
          token_count = EXCLUDED.token_count,
          embedding = EXCLUDED.embedding,
          metadata_json = EXCLUDED.metadata_json
    so a content_hash change overwrites the prior vector cleanly.

    Returns the count of rows actually inserted-or-updated.
    """
    ...

def start_run(conn: psycopg.Connection, embedding_model: str, full_rebuild: bool) -> str:
    """INSERT a new row into ingest_runs with status='running'. Returns the run UUID."""

    conn: psycopg.Connection,
    run_id: str,
    n_documents: int,
    n_chunks_inserted: int,
    n_chunks_skipped: int,

```python
# app/ingest/pipeline.py
import argparse, logging, sys
from pathlib import Path
from app.db import get_conn
from app.config import get_settings
from app.ingest.loader import load_forum_posts
from app.ingest.chunker import chunk_document
from app.embeddings.factory import get_embedder
from app.ingest.writer import (
    upsert_document, chunks_to_embed, upsert_chunks,
    start_run, complete_run, fail_run, truncate_all,
)

def main(argv: list[str] | None = None) -> int:
    parser = argparse.ArgumentParser(prog="app.ingest.pipeline")
    parser.add_argument(
        "--full-rebuild", action="store_true",
        help="Truncate chunks and documents before ingesting (D-04 escape hatch).",
    )
    parser.add_argument(
        "--forum-dir", default="raw_data/forum_posts",
        help="Directory of forum .txt files.",
    )
    args = parser.parse_args(argv)

    settings = get_settings()
    embedder = get_embedder()  # raises if EMBEDDING_MODEL is unimplemented

    with get_conn() as conn:
        run_id = start_run(conn, embedder.model, args.full_rebuild)
        try:
            if args.full_rebuild:
                truncate_all(conn)

            raw_docs = load_forum_posts(Path(args.forum_dir))
            total_inserted = 0
            total_skipped = 0

            for raw_doc in raw_docs:
                doc_id = upsert_document(conn, raw_doc)
                chunks = chunk_document(raw_doc)
                to_embed, to_skip = chunks_to_embed(conn, doc_id, chunks, embedder.model)
                total_skipped += len(to_skip)
                if to_embed:
                    result = embedder.embed_documents([c.text for c in to_embed])
                    upsert_chunks(conn, doc_id, to_embed, result.vectors, embedder.model)
                    total_inserted += len(to_embed)
                conn.commit()  # commit per document so a mid-run crash leaves partial valid state

            complete_run(conn, run_id, len(raw_docs), total_inserted, total_skipped)
            conn.commit()
            print(f"OK: {total_inserted} chunks inserted, {total_skipped} skipped across {len(raw_docs)} documents.")
            return 0
        except Exception as e:
            conn.rollback()
            with get_conn() as fresh:
                fail_run(fresh, run_id, repr(e))
                fresh.commit()
            print(f"FAILED: {e!r}", file=sys.stderr)
            raise

if __name__ == "__main__":
    sys.exit(main())
```

Algorithm notes:

- Idempotency is enforced both by the application-level `chunks_to_embed` check AND by the DB-level `UNIQUE (document_id, chunk_index, embedding_model)` constraint. The application check avoids issuing embedding API calls for unchanged text; the DB constraint is the safety net.
- All INSERT/SELECT/UPDATE statements use psycopg's `%s` parameter placeholders. NEVER f-string text/vectors into SQL.
- `register_vector(conn)` is called inside `get_conn()` (Plan 01); vectors are passed as Python lists and adapted to `vector(1536)` by the pgvector psycopg adapter.
- Use `psycopg.connect(...)` as a context manager — psycopg3 commits on clean exit, rolls back on exception. The pipeline explicitly commits per document for incremental durability.
